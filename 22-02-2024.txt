class Product():
    def __init__(self,code,name,supplier,price):
        self.code = code
        self.name = name
        self.supplier = supplier
        self.price = price
    def info(self):
        print(f" code : {self.code} name : {self.name}")

laptop= Product(1,"hp 11 gen","hp",70000)
laptop.info()
mobile = Product(2,"Samsung 200px ","Samsung",45000)
mobile.info()
